#[acmd_script( agent = "captain", script = "sound_dash", category = ACMD_SOUND, low_priority )]
unsafe fn sound_dash(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 4.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_captain_metal"));
    }
    wait(lua_state, 1.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_captain_dash_start"));
        SET_PLAY_INHIVIT(agent, Hash40::new("se_captain_dash_start"), 20);
    }
    wait(lua_state, 15.0);
    if is_excute(agent) {
        PLAY_STEP(agent, Hash40::new("se_captain_step_right_l"));
    }
    wait(lua_state, 3.0);
    if is_excute(agent) {
        PLAY_STEP(agent, Hash40::new("se_captain_step_left_l"));
    }
}

#[acmd_script( agent = "peach", script = "game_catchpull", category = ACMD_GAME, low_priority )]
unsafe fn game_catchpull(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if IS_EXIST_ARTICLE(fighter, *FIGHTER_PEACH_GENERATE_ARTICLE_KINOPIO) {
        if is_excute(fighter) {
            ArticleModule::remove_exist(boma, *FIGHTER_PEACH_GENERATE_ARTICLE_KINOPIO);
        }
    }
    if is_excute(fighter) {
        ArticleModule::generate_article(boma, *FIGHTER_PEACH_GENERATE_ARTICLE_KINOPIO, false, -1);
        ArticleModule::change_motion(boma, *FIGHTER_PEACH_GENERATE_ARTICLE_KINOPIO, Hash40::new("catch_pull"), false, -1.0);
    }
}

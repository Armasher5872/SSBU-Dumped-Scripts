#[acmd_script( agent = "master", script = "sound_slipescapef", category = ACMD_SOUND, low_priority )]
unsafe fn sound_slipescapef(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 3.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_master_dash_start"));
    }
    frame(agent.lua_state_agent, 10.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_master_escapef"));
    }
    frame(agent.lua_state_agent, 21.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_master_step_left_m"));
    }
    frame(agent.lua_state_agent, 25.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_master_step_right_m"));
    }
}

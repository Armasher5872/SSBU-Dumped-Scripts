#[acmd_script( agent = "ryu", script = "sound_specialairsstart", category = ACMD_SOUND, low_priority )]
unsafe fn sound_specialairsstart(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 1.0);
    if !WorkModule::is_flag(boma, *FIGHTER_RYU_STATUS_WORK_ID_SPECIAL_COMMON_FLAG_COMMAND) {
        wait(lua_state, 7.0);
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("vc_ryu_special_s02"));
        }
        else{
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("se_ryu_command_success"));
        }
        wait(lua_state, 7.0);
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("vc_ryu_special_s02_command"));
        }
    }
}

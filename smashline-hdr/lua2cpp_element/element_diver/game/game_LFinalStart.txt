#[acmd_script( agent = "element_diver", script = "game_lfinalstart", category = ACMD_GAME, low_priority )]
unsafe fn game_lfinalstart(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        VisibilityModule::set_whole(boma, false);
    }
    frame(lua_state, 1.0);
    if is_excute(agent) {
        VisibilityModule::set_whole(boma, true);
    }
    frame(lua_state, 10.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *WEAPON_ELEMENT_DIVER_INSTANCE_WORK_ID_FLAG_FINAL_START_GROUND_ENABLE);
    }
    frame(lua_state, 120.0);
    if is_excute(agent) {
        VisibilityModule::set_whole(boma, false);
    }
}

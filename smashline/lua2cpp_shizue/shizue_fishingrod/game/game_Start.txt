#[acmd_script( agent = "shizue_fishingrod", script = "game_start", category = ACMD_GAME, low_priority )]
unsafe fn game_start(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 0.0);
    if macros::is_excute(agent) {
        MotionModule::set_rate(agent.module_accessor, 1.25);
    }
    frame(agent.lua_state_agent, 20.0);
    if macros::is_excute(agent) {
        MotionModule::set_rate(agent.module_accessor, 1);
    }
}

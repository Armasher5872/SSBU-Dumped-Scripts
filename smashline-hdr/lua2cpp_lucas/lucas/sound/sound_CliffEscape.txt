#[acmd_script( agent = "lucas", script = "sound_cliffescape", category = ACMD_SOUND, low_priority )]
unsafe fn sound_cliffescape(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 3.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_lucas_dash_start"));
    }
    wait(lua_state, 8.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_lucas_escape"));
    }
    wait(lua_state, 21.0);
    if is_excute(agent) {
        PLAY_STEP(agent, Hash40::new("se_lucas_step_right_m"));
    }
    wait(lua_state, 11.0);
    if is_excute(agent) {
        PLAY_STEP(agent, Hash40::new("se_lucas_step_left_m"));
    }
}

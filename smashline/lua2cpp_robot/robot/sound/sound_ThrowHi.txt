#[acmd_script( agent = "robot", script = "sound_throwhi", category = ACMD_SOUND, low_priority)]
unsafe fn sound_throwhi(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 7.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_common_throw_01"));
    }
    frame(fighter.lua_state_agent, 15.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_robot_jump01"));
        macros::PLAY_SEQUENCE(fighter, Hash40::new("seq_robot_rnd_attack"));
    }
    wait(fighter.lua_state_agent, 36.0);
    if macros::is_excute(fighter) {
        macros::PLAY_LANDING_SE(fighter, Hash40::new("se_common_throw_03"));
    }
    wait(fighter.lua_state_agent, 5.0);
    if macros::is_excute(fighter) {
        macros::PLAY_LANDING_SE(fighter, Hash40::new("se_robot_landing02"));
        macros::PLAY_DOWN_SE(fighter, Hash40::new("se_common_down_m_01"));
        macros::PLAY_SE(fighter, Hash40::new("se_common_kick_hit_m"));
    }
}

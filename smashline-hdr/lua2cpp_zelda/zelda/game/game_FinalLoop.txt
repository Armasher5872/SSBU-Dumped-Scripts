            AttackModule::set_final_finish_cut_in(boma, 4, true, true, -1, false);
        }
    }
    wait(lua_state, 2.0);

        AttackModule::clear(boma, 4, false);

            AttackModule::set_final_finish_cut_in(boma, 4, true, true, -1, false);
        }
    }
    wait(lua_state, 2.0);
    if is_excute(fighter) {
        AttackModule::clear(boma, 4, false);
    }
}

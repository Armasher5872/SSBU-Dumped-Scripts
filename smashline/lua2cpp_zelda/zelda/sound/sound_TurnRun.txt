#[acmd_script( agent = "zelda", script = "sound_turnrun", category = ACMD_SOUND, low_priority )]
unsafe fn sound_turnrun(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 6.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_zelda_dash_stop"));
        macros::SET_PLAY_INHIVIT(fighter, Hash40::new("se_zelda_dash_stop"), 10);
    }
    frame(fighter.lua_state_agent, 19.0);
    if macros::is_excute(fighter) {
        macros::PLAY_STEP_FLIPPABLE(fighter, Hash40::new("se_zelda_step_left_l"), Hash40::new("se_zelda_step_right_l"));
    }
}

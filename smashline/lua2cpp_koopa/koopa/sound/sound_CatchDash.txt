#[acmd_script( agent = "koopa", script = "sound_catchdash", category = ACMD_SOUND, low_priority )]
unsafe fn sound_catchdash(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 10.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_koopa_catch"));
        macros::PLAY_LANDING_SE(agent, Hash40::new("se_koopa_landing01"));
    }
    wait(agent.lua_state_agent, 6.0);
    if macros::is_excute(agent) {
        macros::STOP_SE(agent, Hash40::new("se_koopa_catch"));
    }
    wait(agent.lua_state_agent, 34.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP_FLIPPABLE(agent, Hash40::new("se_koopa_step_left_m"), Hash40::new("se_koopa_step_right_m"));
    }
}

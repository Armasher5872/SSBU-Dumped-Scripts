#[acmd_script( agent = "popo", script = "sound_escapef", category = ACMD_SOUND, low_priority )]
unsafe fn sound_escapef(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 5.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_popo_escape"));
    }
    frame(fighter.lua_state_agent, 8.0);
    if macros::is_excute(fighter) {
        macros::PLAY_STEP_FLIPPABLE(fighter, Hash40::new("se_popo_step_right_s"), Hash40::new("se_popo_step_left_s"));
    }
    wait(fighter.lua_state_agent, 8.0);
    if macros::is_excute(fighter) {
        macros::PLAY_STEP_FLIPPABLE(fighter, Hash40::new("se_popo_step_left_s"), Hash40::new("se_popo_step_right_s"));
    }
    wait(fighter.lua_state_agent, 21.0);
    if macros::is_excute(fighter) {
        macros::PLAY_STEP_FLIPPABLE(fighter, Hash40::new("se_popo_step_left_s"), Hash40::new("se_popo_step_right_s"));
    }
}

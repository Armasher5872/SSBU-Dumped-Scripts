#[acmd_script( agent = "miigunner", script = "sound_slipattack", category = ACMD_SOUND, low_priority )]
unsafe fn sound_slipattack(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 18.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_miigunner_swing_l"));
    }
    frame(agent.lua_state_agent, 31.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_miigunner_swing_l"));
    }
    wait(agent.lua_state_agent, 20.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_miigunner_step_right_m"));
    }
    wait(agent.lua_state_agent, 6.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_miigunner_step_left_m"));
    }
}

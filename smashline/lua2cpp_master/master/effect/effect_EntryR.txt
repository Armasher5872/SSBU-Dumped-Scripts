unsafe fn effect_entryr(fighter: &mut L2CAgentBase) {
	if macros::is_excute(fighter) {
		macros::EFFECT(fighter, Hash40::new("master_entry"), Hash40::new("top"), 0, 0, 0, 0, 90, 0, 1, 0, 0, 0, 0, 0, 0, true);
		macros::LAST_EFFECT_SET_RATE(fighter, 1.2);
		macros::FLASH(fighter, 0.5, 0.8, 0.7, 0.8);
		macros::BURN_COLOR(fighter, 0.4, 2, 1.4, 0.9);
		ColorBlendModule::set_disable_camera_depth_influence(fighter.module_accessor, true);
	}
	wait(fighter.lua_state_agent, 1.0);
	if macros::is_excute(fighter) {
		macros::FLASH_FRM(fighter, 35, 0.4, 0.8, 0.7, 0);
		macros::BURN_COLOR_FRAME(fighter, 40, 0.4, 2, 1.4, 0);
	}
	wait(fighter.lua_state_agent, 35.0);
	if macros::is_excute(fighter) {
		macros::COL_NORMAL(fighter);
		macros::BURN_COLOR_NORMAL(fighter);
		ColorBlendModule::set_disable_camera_depth_influence(fighter.module_accessor, false);
	}

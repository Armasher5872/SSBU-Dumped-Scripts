#[acmd_script( agent = "robot", script = "effect_run", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_run(agent: &mut L2CAgentBase) {
    if macros::is_excute(agent) {
        macros::EFFECT_FOLLOW(agent, Hash40::new("robot_nozzle"), Hash40::new("knee1"), 1.2, 0, 0, 90, -90, 0, 1, true);
    }
    frame(agent.lua_state_agent, 1.0);
    if macros::is_excute(agent) {
        macros::FOOT_EFFECT(agent, Hash40::new("robot_run_smoke"), Hash40::new("top"), 0, 0, 0, 0, 0, 0, 1.5, 1, 0, 0, 0, 0, 0, false);
    }
    wait(agent.lua_state_agent, 10.0);
}

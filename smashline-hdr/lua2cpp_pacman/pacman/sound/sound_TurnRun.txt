#[acmd_script( agent = "pacman", script = "sound_turnrun", category = ACMD_SOUND, low_priority )]
unsafe fn sound_turnrun(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 1.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_pacman_dash_turn"));
        SET_PLAY_INHIVIT(agent, Hash40::new("se_pacman_dash_stop"), 30);
    }
    wait(lua_state, 24.0);
    if is_excute(agent) {
        PLAY_STEP_FLIPPABLE(agent, Hash40::new("se_pacman_step_left_l"), Hash40::new("se_pacman_step_right_l"));
    }
    wait(lua_state, 11.0);
    if is_excute(agent) {
        PLAY_STEP_FLIPPABLE(agent, Hash40::new("se_pacman_step_right_l"), Hash40::new("se_pacman_step_left_l"));
    }
}

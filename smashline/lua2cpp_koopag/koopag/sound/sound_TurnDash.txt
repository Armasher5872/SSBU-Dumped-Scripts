    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_koopag_dash_start"));
    }
    wait(agent.lua_state_agent, 11.0);

        macros::PLAY_STEP(agent, Hash40::new("se_koopag_step_right_m"));
    }

    }
}

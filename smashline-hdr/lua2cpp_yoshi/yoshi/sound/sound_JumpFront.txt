unsafe extern "C" fn sound_jumpfront(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 1.0);
    if WorkModule::is_flag(boma, *FIGHTER_STATUS_JUMP_FLAG_GIMMICK_SPRING_JUMP_FROM_RING) {
        wait(lua_state, 3.0);
        if is_excute(agent) {
            PLAY_SEQUENCE(agent, Hash40::new("seq_yoshi_rnd_jump01"));
        }
        else{
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("se_yoshi_jump01"));
        }
        wait(lua_state, 3.0);
        if is_excute(agent) {
            PLAY_SEQUENCE(agent, Hash40::new("seq_yoshi_rnd_jump01"));
        }
    }
}

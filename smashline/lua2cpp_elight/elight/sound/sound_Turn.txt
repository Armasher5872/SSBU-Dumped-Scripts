#[acmd_script( agent = "elight", script = "sound_turn", category = ACMD_SOUND, low_priority )]
unsafe fn sound_turn(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 2.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_elight_step_right_s"));
    }
    frame(agent.lua_state_agent, 15.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_elight_step_left_s"));
    }
}

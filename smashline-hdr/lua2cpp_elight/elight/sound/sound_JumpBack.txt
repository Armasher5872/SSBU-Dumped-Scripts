#[acmd_script( agent = "elight", script = "sound_jumpback", category = ACMD_SOUND, low_priority )]
unsafe fn sound_jumpback(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 2.0);
    if WorkModule::is_flag(boma, *FIGHTER_STATUS_JUMP_FLAG_GIMMICK_SPRING_JUMP_FROM_RING) {
        if is_excute(agent) {
            PLAY_SEQUENCE(agent, Hash40::new("seq_elight_rnd_jump"));
        }
        else{
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("se_elight_jump01"));
            PLAY_SEQUENCE(agent, Hash40::new("seq_elight_rnd_jump"));
        }
    }
}
}

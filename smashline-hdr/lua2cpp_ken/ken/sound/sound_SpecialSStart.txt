    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 1.0);
    if !WorkModule::is_flag(boma, *FIGHTER_RYU_STATUS_WORK_ID_SPECIAL_COMMON_FLAG_COMMAND) {
        wait(lua_state, 7.0);
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("vc_ken_special_s01"));
        }
        else{
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("se_ken_command_success"));
        }
        wait(lua_state, 2.0);
        if is_excute(agent) {
            PLAY_SE(agent, Hash40::new("vc_ken_special_s01_command"));
        }
    }
}

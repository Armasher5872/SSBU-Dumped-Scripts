#[acmd_script( agent = "rosetta", script = "sound_ladderup", category = ACMD_SOUND, low_priority )]
unsafe fn sound_ladderup(agent: &mut L2CAgentBase) {
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_rosetta_step_right_m"));
    }
    wait(agent.lua_state_agent, 5.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_rosetta_step_left_m"));
    }
    wait(agent.lua_state_agent, 5.0);
}

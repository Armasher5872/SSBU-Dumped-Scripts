#[acmd_script( agent = "pit", script = "sound_specialsstart", category = ACMD_SOUND, low_priority )]
unsafe fn sound_specialsstart(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 4.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_pit_special_s01"));
        macros::PLAY_SEQUENCE(agent, Hash40::new("seq_pit_rnd_special_s01"));
    }
    frame(agent.lua_state_agent, 14.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_pit_special_s02"));
    }
}

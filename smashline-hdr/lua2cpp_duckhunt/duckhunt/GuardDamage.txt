    let boma = agent.boma();
    if is_excute(agent) {
        HIT_NODE(agent, Hash40::new("duckhip"), *HIT_STATUS_XLU);
    }
    frame(lua_state, 21.0);
    if is_excute(agent) {
        HIT_NODE(agent, Hash40::new("duckhip"), *HIT_STATUS_NORMAL);
    }
}

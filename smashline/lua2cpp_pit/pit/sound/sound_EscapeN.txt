#[acmd_script( agent = "pit", script = "sound_escapen", category = ACMD_SOUND, low_priority )]
unsafe fn sound_escapen(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 1.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_pit_escape"));
    }
    wait(agent.lua_state_agent, 32.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_pit_step_right_m"));
    }
}

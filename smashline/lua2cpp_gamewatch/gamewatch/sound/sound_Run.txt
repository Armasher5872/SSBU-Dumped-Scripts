#[acmd_script( agent = "gamewatch", script = "sound_run", category = ACMD_SOUND, low_priority )]
unsafe fn sound_run(agent: &mut L2CAgentBase) {
    wait_loop_sync_mot();
    frame(agent.lua_state_agent, 0.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_gamewatch_step_right_m"));
    }
    frame(agent.lua_state_agent, 10.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_gamewatch_step_left_m"));
    }
    frame(agent.lua_state_agent, 20.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_gamewatch_step_right_m"));
    }
    frame(agent.lua_state_agent, 30.0);
    macros::PLAY_STEP(agent, Hash40::new("se_gamewatch_step_left_m"));
}

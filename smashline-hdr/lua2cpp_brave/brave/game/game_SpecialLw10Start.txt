#[acmd_script( agent = "brave", script = "game_speciallw10start", category = ACMD_GAME, low_priority )]
unsafe fn game_speciallw10start(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 1.0);
    FT_MOTION_RATE(agent, 0.8);
    frame(lua_state, 10.0);
    FT_MOTION_RATE(agent, 0.65);
    frame(lua_state, 15.0);
    if is_excute(agent) {
        PhysicsModule::set_swing_rate(boma, 0);
    }
    frame(lua_state, 20.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_BRAVE_STATUS_SPECIAL_LW_START_FLAG_ENABLE_STEEL);
    }
}

#[acmd_script( agent = "yoshi", script = "sound_dash", category = ACMD_SOUND, low_priority )]
unsafe fn sound_dash(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 5.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_yoshi_dash_start"));
        macros::SET_PLAY_INHIVIT(agent, Hash40::new("se_yoshi_dash_start"), 20);
    }
    wait(agent.lua_state_agent, 11.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_yoshi_step_right_l"));
    }
    wait(agent.lua_state_agent, 2.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_yoshi_step_left_l"));
    }
}

#[acmd_script( agent = "plizardon", script = "sound_jumpback", category = ACMD_SOUND, low_priority )]
unsafe fn sound_jumpback(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 3.0);
    if WorkModule::is_flag(fighter.module_accessor, *FIGHTER_STATUS_JUMP_FLAG_GIMMICK_SPRING_JUMP_FROM_RING) {
        wait(fighter.lua_state_agent, 2.0);
        if macros::is_excute(fighter) {
            macros::PLAY_SE(fighter, Hash40::new("se_plizardon_wing"));
        }
        else{
        if macros::is_excute(fighter) {
            macros::PLAY_SE(fighter, Hash40::new("se_plizardon_jump01"));
        }
        wait(fighter.lua_state_agent, 2.0);
        if macros::is_excute(fighter) {
            macros::PLAY_SE(fighter, Hash40::new("se_plizardon_wing"));
        }
    }
}
}

#[acmd_script( agent = "falco", script = "sound_passivewall" , category = ACMD_SOUND , low_priority)]
unsafe fn sound_passivewall(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::STOP_SE(fighter, Hash40::new("se_common_futtobi_s"));
        macros::STOP_SE(fighter, Hash40::new("se_common_futtobi_m"));
        macros::STOP_SE(fighter, Hash40::new("se_common_futtobi_l"));
        macros::PLAY_SE(fighter, Hash40::new("vc_falco_passive"));
        macros::PLAY_SE(fighter, Hash40::new("se_falco_landing01"));
    }
}

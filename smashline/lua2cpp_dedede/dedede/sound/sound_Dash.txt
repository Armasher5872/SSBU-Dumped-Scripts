    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_dedede_dash_start"));
        macros::SET_PLAY_INHIVIT(fighter, Hash40::new("se_dedede_dash_start"), 15);

    }
}

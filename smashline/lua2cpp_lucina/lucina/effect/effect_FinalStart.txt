#[acmd_script( agent = "lucina", script = "effect_finalstart", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_finalstart(agent: &mut L2CAgentBase) {
    if macros::is_excute(agent) {
        EffectModule::req_screen(agent.module_accessor, Hash40::new("bg_lucina_final"), false, false, false);
    }
    frame(agent.lua_state_agent, 26.0);
    if macros::is_excute(agent) {
        macros::EFFECT_FLW_POS(agent, Hash40::new("lucina_final_light"), Hash40::new("haver"), 0, 12, 0.0, 0, 0, 0, 1.5, true);
        macros::EFFECT_FLW_POS(agent, Hash40::new("lucina_final_lensflare"), Hash40::new("haver"), 0, 12, 0.0, 0, 0, 0, 1.5, true);
    }
    frame(agent.lua_state_agent, 30.0);
    if macros::is_excute(agent) {
        macros::EFFECT_FOLLOW(agent, Hash40::new("lucina_final_sword_aura"), Hash40::new("haver"), 0, 0, 0, 0, 0, 0, 1, true);
    }
    frame(agent.lua_state_agent, 45.0);
    if macros::is_excute(agent) {
        macros::EFFECT_DETACH_KIND(agent, Hash40::new("lucina_final_light"), -1);
        macros::EFFECT_DETACH_KIND(agent, Hash40::new("lucina_final_lensflare"), -1);
    }
}

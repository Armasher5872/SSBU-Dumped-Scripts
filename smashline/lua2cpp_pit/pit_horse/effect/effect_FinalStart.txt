#[acmd_script( agent = "pit_horse", script = "effect_finalstart", category = ACMD_EFFECT, low_priority)]
unsafe fn effect_finalstart(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::EFFECT_FLW_POS(fighter, Hash40::new("pit_final_enter"), Hash40::new("bust"), 0, 0, 0, 0, 0, 0, 1, true);
    }
    frame(fighter.lua_state_agent, 145.0);
    if macros::is_excute(fighter) {
        macros::EFFECT_OFF_KIND(fighter, Hash40::new("pit_final_enter"), false, true);
    }
    frame(fighter.lua_state_agent, 160.0);
    if macros::is_excute(fighter) {
        macros::EFFECT(fighter, Hash40::new("pit_final_flash"), Hash40::new("bust"), 0, 0, 0, 0, 0, 0, 80, 0, 0, 0, 0, 0, 0, true);
    }
}

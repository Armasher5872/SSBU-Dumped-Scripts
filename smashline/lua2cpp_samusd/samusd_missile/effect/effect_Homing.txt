#[acmd_script( agent = "samusd_missile", script = "effect_homing", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_homing(fighter: &mut L2CAgentBase) {

        macros::EFFECT_FOLLOW(fighter, Hash40::new("samusd_missile_homing"), Hash40::new("top"), 0, 0, 0, 0, 0, 0, 1, true);
    }
    frame(fighter.lua_state_agent, 60.0);

        macros::EFFECT_OFF_KIND(fighter, Hash40::new("samusd_missile_homing"), false, true);
    }
}

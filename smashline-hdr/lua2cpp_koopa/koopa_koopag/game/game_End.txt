#[acmd_script( agent = "koopa_koopag", script = "game_end", category = ACMD_GAME, low_priority )]
unsafe fn game_end(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        ControlModule::set_rumble(boma, Hash40::new("rbkind_14_end_sp"), 0, true, *BATTLE_OBJECT_ID_INVALID as u32);
    }
    frame(lua_state, 30.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *WEAPON_KOOPA_KOOPAG_INSTANCE_WORK_ID_FLAG_END_UP);
    }
}

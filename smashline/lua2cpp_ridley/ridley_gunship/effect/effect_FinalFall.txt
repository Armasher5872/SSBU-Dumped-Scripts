#[acmd_script( agent = "ridley_gunship", script = "effect_finalfall", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_finalfall(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::EFFECT_OFF_KIND(fighter, Hash40::new("ridley_final_explosion"), true, true);
        macros::EFFECT_FOLLOW(fighter, Hash40::new("ridley_final_smoke"), Hash40::new("gearr"), 2, 0, 0, 0, 0, 0, 1, true);
    }
    frame(fighter.lua_state_agent, 80.0);
    if macros::is_excute(fighter) {
        macros::EFFECT_FOLLOW(fighter, Hash40::new("ridley_final_bomb"), Hash40::new("hip"), 0, 0, 0, 0, 0, 0, 2.5, true);
    }
}

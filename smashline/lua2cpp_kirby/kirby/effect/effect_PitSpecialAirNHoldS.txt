#[acmd_script( agent = "kirby", script = "effect_pitspecialairnholds", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_pitspecialairnholds(agent: &mut L2CAgentBase) {
    if macros::is_excute(agent) {
        macros::EFFECT_OFF_KIND(agent, Hash40::new("pit_pa_hold_bow"), true, true);
        macros::EFFECT_OFF_KIND(agent, Hash40::new("pit_pa_hold_bow2"), true, true);
    }
    if get_value_float(agent, *SO_VAR_FLOAT_LR) < 0.0 {
        if macros::is_excute(agent) {
            macros::EFFECT_FOLLOW(agent, Hash40::new("pit_pa_hold_bow"), Hash40::new("havel"), 1, 2.5, 1, -1, -20, 180, 1, true);
        }
        else{
        if macros::is_excute(agent) {
            macros::EFFECT_FOLLOW(agent, Hash40::new("pit_pa_hold_bow"), Hash40::new("havel"), -0.85, -1.5, 1.1, 1, -20, -2, 1, true);
        }
    }
}

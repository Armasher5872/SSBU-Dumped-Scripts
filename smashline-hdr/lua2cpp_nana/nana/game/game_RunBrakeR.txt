#[acmd_script( agent = "nana", script = "game_runbraker", category = ACMD_GAME, low_priority )]
unsafe fn game_runbraker(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 12.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_STATUS_RUN_BRAKE_FLAG_STOP);
        WorkModule::off_flag(boma, *FIGHTER_STATUS_RUN_BRAKE_FLAG_TURN_RUN);
    }
}

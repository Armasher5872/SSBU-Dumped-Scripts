#[acmd_script( agent = "toonlink", script = "game_specialnstart", category = ACMD_GAME, low_priority )]
unsafe fn game_specialnstart(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        ArticleModule::generate_article(boma, *FIGHTER_TOONLINK_GENERATE_ARTICLE_BOW, false, -1);
        ArticleModule::generate_article(boma, *FIGHTER_TOONLINK_GENERATE_ARTICLE_BOWARROW, false, -1);
    }
    frame(lua_state, 18.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_LINK_STATUS_BOW_FLAG_CHARGE);
    }
}

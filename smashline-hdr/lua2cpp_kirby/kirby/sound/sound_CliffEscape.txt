#[acmd_script( agent = "kirby", script = "sound_cliffescape", category = ACMD_SOUND, low_priority )]
unsafe fn sound_cliffescape(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 5.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_kirby_dash_start"));
    }
    frame(lua_state, 16.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_kirby_escape"));

    frame(lua_state, 30.0);

        PLAY_SE(agent, Hash40::new("se_kirby_dash_stop"));
    }
}

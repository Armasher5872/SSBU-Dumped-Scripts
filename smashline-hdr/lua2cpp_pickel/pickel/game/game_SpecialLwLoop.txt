#[acmd_script( agent = "pickel", script = "game_speciallwloop", category = ACMD_GAME, low_priority )]
unsafe fn game_speciallwloop(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_PICKEL_INSTANCE_WORK_ID_FLAG_REQUEST_REMOVE_HAVE_CRAFT_WEAPON);
    }
    wait(lua_state, 1.0);
    WorkModule::on_flag(boma, *FIGHTER_PICKEL_STATUS_SPECIAL_LW_FLAG_GENERATE_STONE);
}

        macros::STOP_SE(agent, Hash40::new("se_koopajr_special_s02"));
        macros::STOP_SE(agent, Hash40::new("se_koopajr_special_s02_03"));
    }

            macros::PLAY_STATUS(agent, Hash40::new("se_koopajr_special_s02_01"));
        }
    }
}

#[acmd_script( agent = "wario", script = "sound_turnrun", category = ACMD_SOUND, low_priority )]
unsafe fn sound_turnrun(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 5.0);
    if macros::is_excute(agent) {
        macros::STOP_SE(agent, Hash40::new("se_wario_dash_stop"));
        macros::PLAY_SE(agent, Hash40::new("se_wario_dash_turn"));
        macros::SET_PLAY_INHIVIT(agent, Hash40::new("se_wario_dash_turn"), 15);
    }
}

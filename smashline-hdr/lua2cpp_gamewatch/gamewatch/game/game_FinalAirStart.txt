#[acmd_script( agent = "gamewatch", script = "game_finalairstart", category = ACMD_GAME, low_priority )]
unsafe fn game_finalairstart(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        SLOW_OPPONENT(agent, 10, 60);
        FT_REMOVE_FINAL_AURA(agent, true);
        FT_SET_FINAL_FEAR_FACE(agent, 70);
        PostureModule::scale(boma, 1, 0);
        0x102320(-1174958658, 2.1);
        CAM_ZOOM_IN_arg5(0, 0);
    }
    frame(lua_state, 5.0);
    if is_excute(agent) {
        FT_START_CUTIN(agent);
    }
    frame(lua_state, 12.0);
    if is_excute(agent) {
        CAM_ZOOM_OUT(agent);
        PostureModule::scale(boma, 15, 0);
        0x102320(-1174958658, 3.5);
        CAM_ZOOM_IN_arg5(0, 0);
    }
    if sv_animcmd::get_value_float(agent, *SO_VAR_FLOAT_LR) < 0.0 {
        if is_excute(agent) {
            camera!(agent, *MA_MSC_CMD_CAMERA_CAM_OFFSET, -10, 5);
        }
        else{
        if is_excute(agent) {
            camera!(agent, *MA_MSC_CMD_CAMERA_CAM_OFFSET, 10, 5);
        }
    }
}
frame(lua_state, 60.0);
if is_excute(agent) {
    CAM_ZOOM_OUT(agent);
}
}

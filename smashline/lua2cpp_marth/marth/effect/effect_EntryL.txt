#[acmd_script( agent = "marth", script = "effect_entryl", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_entryl(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::EFFECT(fighter, Hash40::new("marth_entry"), Hash40::new("top"), 0, 0, 0, 0, 270, 0, 1, 0, 0, 0, 0, 0, 0, true);
        macros::FLASH(fighter, 0.8, 0.8, 0.5, 0.8);
        macros::BURN_COLOR(fighter, 2.6, 2.6, 0.85, 0.9);
        ColorBlendModule::set_disable_camera_depth_influence(fighter.module_accessor, true);
    }
    wait(fighter.lua_state_agent, 1.0);
    if macros::is_excute(fighter) {
        macros::FLASH_FRM(fighter, 35, 0.8, 0.8, 0.4, 0);
        macros::BURN_COLOR_FRAME(fighter, 35, 2.6, 2.6, 0.85, 0);
    }
    wait(fighter.lua_state_agent, 35.0);
    if macros::is_excute(fighter) {
        macros::COL_NORMAL(fighter);
        macros::BURN_COLOR_NORMAL(fighter);
        ColorBlendModule::set_disable_camera_depth_influence(fighter.module_accessor, false);
    }
}

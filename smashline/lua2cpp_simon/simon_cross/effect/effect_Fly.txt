            macros::EFFECT_FOLLOW_NO_STOP(agent, Hash40::new("simon_cross"), Hash40::new("rot"), 0, 0, 0, 0, -90, 0, 1, true);
        }
        else{
        if macros::is_excute(agent) {
            macros::EFFECT_FOLLOW_NO_STOP(agent, Hash40::new("simon_cross"), Hash40::new("rot"), 0, 0, 0, 0, 90, 0, 1, true);
        }
    }
}
}

#[acmd_script( agent = "demon", script = "sound_turnrun", category = ACMD_SOUND, low_priority )]
unsafe fn sound_turnrun(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 1.0);
    if macros::is_excute(fighter) {
        macros::STOP_SE(fighter, Hash40::new("se_demon_dash_stop"));
    }
    frame(fighter.lua_state_agent, 4.0);
    if macros::is_excute(fighter) {
        macros::PLAY_STATUS(fighter, Hash40::new("se_demon_dash_turn"));
        macros::SET_PLAY_INHIVIT(fighter, Hash40::new("se_demon_dash_stop"), 20);
    }
}

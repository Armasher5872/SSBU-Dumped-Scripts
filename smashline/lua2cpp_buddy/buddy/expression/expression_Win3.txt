#[acmd_script( agent = "buddy", script = "expression_win3", category = ACMD_EXPRESSION, low_priority )]
unsafe fn expression_win3(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        ArticleModule::remove_exist(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_PIECE, ArticleOperationTarget(*ARTICLE_OPE_TARGET_ALL));
        ArticleModule::remove_exist(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_STRINGS, ArticleOperationTarget(*ARTICLE_OPE_TARGET_ALL));
        ArticleModule::remove_exist(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_HORN, ArticleOperationTarget(*ARTICLE_OPE_TARGET_ALL));
    }
    if !macros::IS_EXIST_ARTICLE(fighter, *FIGHTER_BUDDY_GENERATE_ARTICLE_STRINGS) {
        if macros::is_excute(fighter) {
            ArticleModule::generate_article(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_STRINGS, false, -1);
        }
    }
    if macros::is_excute(fighter) {
        ArticleModule::change_motion(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_STRINGS, Hash40::new("win_3"), false, -1.0);
    }
    if !macros::IS_EXIST_ARTICLE(fighter, *FIGHTER_BUDDY_GENERATE_ARTICLE_HORN) {
        if macros::is_excute(fighter) {
            ArticleModule::generate_article(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_HORN, false, -1);
        }
    }
    if macros::is_excute(fighter) {
        ArticleModule::change_motion(fighter.module_accessor, *FIGHTER_BUDDY_GENERATE_ARTICLE_HORN, Hash40::new("win_3"), false, -1.0);
    }
}

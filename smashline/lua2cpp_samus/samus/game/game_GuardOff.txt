        macros::HIT_NODE(agent, Hash40::new("head"), *HIT_STATUS_XLU);
        macros::HIT_NODE(agent, Hash40::new("shoulderpadl"), *HIT_STATUS_XLU);
        macros::HIT_NODE(agent, Hash40::new("armr"), *HIT_STATUS_XLU);

        macros::HIT_NODE(agent, Hash40::new("shoulderpadl"), *HIT_STATUS_NORMAL);

}

        macros::STOP_SE(fighter, Hash40::new("se_wolf_final02"));
        PLAY_SE_NO_3D(Hash40::new("se_wolf_final03"));
    }
    frame(fighter.lua_state_agent, 40.0);

        PLAY_SE_NO_3D(Hash40::new("se_wolf_final_noise_start"));
    }

            PLAY_SE_NO_3D_FORCE_AUTO_PITCH(Hash40::new("vc_wolf_final03"));
        }

        }
    }
}

    PLAY_SE_NO_3D(Hash40::new("se_wolf_final_noise_end"));
}
}

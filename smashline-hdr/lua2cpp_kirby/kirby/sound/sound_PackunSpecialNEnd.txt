#[acmd_script( agent = "kirby", script = "sound_packunspecialnend", category = ACMD_SOUND, low_priority )]
unsafe fn sound_packunspecialnend(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if WorkModule::is_flag(boma, *FIGHTER_PACKUN_STATUS_SPECIAL_N_WORK_FLAG_END_PREVIOUS_SHOOT) {
        frame(lua_state, 1.0);
        if is_excute(agent) {
            STOP_SE(agent, Hash40::new("se_packun_special_n03"));
        }
    }
    frame(lua_state, 3.0);
    if is_excute(agent) {
        PLAY_SE(agent, Hash40::new("se_packun_special_n05"));
    }
}

#[acmd_script( agent = "fox", script = "game_appealsr", category = ACMD_GAME, low_priority )]
unsafe fn game_appealsr(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 2.0);
    if is_excute(agent) {
        ArticleModule::remove_exist(boma, *FIGHTER_FOX_GENERATE_ARTICLE_BLASTER, ArticleOperationTarget(*ARTICLE_OPE_TARGET_ALL));
        ArticleModule::generate_article(boma, *FIGHTER_FOX_GENERATE_ARTICLE_BLASTER, true, -1);
    }
    frame(lua_state, 71.0);
    if is_excute(agent) {
        ArticleModule::remove_exist(boma, *FIGHTER_FOX_GENERATE_ARTICLE_BLASTER, ArticleOperationTarget(*ARTICLE_OPE_TARGET_ALL));
    }
}

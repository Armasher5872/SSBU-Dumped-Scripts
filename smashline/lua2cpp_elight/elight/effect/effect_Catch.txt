#[acmd_script( agent = "elight", script = "effect_catch" , category = ACMD_EFFECT , low_priority)]
unsafe fn effect_catch(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::EFFECT_FOLLOW(fighter, Hash40::new("elight_catch_hand"), Hash40::new("handl"), 0.5, 0, 0, 0, 0, 0, 1, true);
        macros::EFFECT_FOLLOW(fighter, Hash40::new("elight_catch_hand"), Hash40::new("handr"), 0.5, 0, 0, 0, 0, 0, 1, true);
    }
    frame(fighter.lua_state_agent, 16.0);
    if macros::is_excute(fighter) {
        macros::EFFECT_OFF_KIND(fighter, Hash40::new("elight_catch_hand"), false, true);
    }
}

        macros::WHOLE_HIT(fighter, *HIT_STATUS_XLU);
        macros::CHECK_VALID_FINAL_START_CAMERA(fighter, 0, 7, 20, 21, 0, 0);
        macros::SLOW_OPPONENT(fighter, 4, 45);
    }
    WorkModule::is_flag(fighter.module_accessor, *FIGHTER_INSTANCE_WORK_ID_FLAG_DISABLE_FINAL_START_CAMERA);
    if(methodlib::L2CValue::operator==(lib::L2CValueconst&)const(false, false)){
        frame(fighter.lua_state_agent, 10.0);
        if macros::is_excute(fighter) {
            macros::FT_SET_FINAL_FEAR_FACE(fighter, 60);
        }
    }
    if get_value_float(fighter, *SO_VAR_FLOAT_LR) < 0.0 {
        if macros::is_excute(fighter) {
            macros::REQ_FINAL_START_CAMERA(fighter, Hash40::new("d04finalstart02.nuanmb"), false);
        }
        else{
        if macros::is_excute(fighter) {
            camera!(fighter, *MA_MSC_CMD_CAMERA_CAM_RECT, 0, 10, 0, 0);
            PostureModule::scale(fighter.module_accessor, 5, 0);
            0x137f20(-692780195, 2);
            CAM_ZOOM_IN_arg5(0, 0);
            macros::FT_START_CUTIN(fighter);
        }
        frame(fighter.lua_state_agent, 25.0);
        if macros::is_excute(fighter) {
            macros::CAM_ZOOM_OUT(fighter);
        }
        else{
        if macros::is_excute(fighter) {
            macros::REQ_FINAL_START_CAMERA(fighter, Hash40::new("d04finalstart.nuanmb"), false);
        }
        if macros::is_excute(fighter) {
            macros::FT_START_CUTIN(fighter);
        }
    }
}

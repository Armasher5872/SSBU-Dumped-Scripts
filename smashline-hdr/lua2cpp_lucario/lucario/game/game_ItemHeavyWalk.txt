#[acmd_script( agent = "lucario", script = "game_itemheavywalk", category = ACMD_GAME, low_priority )]
unsafe fn game_itemheavywalk(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_STATUS_ITEM_LIFT_WORK_FLAG_STOP);
    }
    frame(lua_state, 6.0);
    if is_excute(agent) {
        WorkModule::off_flag(boma, *FIGHTER_STATUS_ITEM_LIFT_WORK_FLAG_STOP);
    }
    frame(lua_state, 26.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_STATUS_ITEM_LIFT_WORK_FLAG_STOP);
    }
}

#[acmd_script( agent = "pikmin_pikmin", script = "effect_flashing", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_flashing(agent: &mut L2CAgentBase) {
    for _ in 0..6 {
    if macros::is_excute(agent) {
        macros::FLASH(agent, 1, 1, 1, 0.588);
    }
    wait(agent.lua_state_agent, 1.0);
    if macros::is_excute(agent) {
        macros::FLASH_FRM(agent, 5, 0, 0, 0, 0.392);
    }
    wait(agent.lua_state_agent, 5.0);
    if macros::is_excute(agent) {
        macros::FLASH_FRM(agent, 8, 0, 0, 0, 0);
    }
    wait(agent.lua_state_agent, 8.0);
    if macros::is_excute(agent) {
        macros::COL_NORMAL(agent);
    }
    wait(agent.lua_state_agent, 1.0);
}

    macros::FLASH(agent, 1, 1, 1, 0.588);
}

}

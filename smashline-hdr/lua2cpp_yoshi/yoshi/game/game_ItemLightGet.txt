#[acmd_script( agent = "yoshi", script = "game_itemlightget", category = ACMD_GAME, low_priority )]
unsafe fn game_itemlightget(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 1.0);
    if is_excute(agent) {
        HIT_NODE(agent, Hash40::new("snout"), *HIT_STATUS_XLU);
    }
    frame(lua_state, 2.0);
    if is_excute(agent) {
        ItemModule::pickup_item(boma, *ITEM_SIZE_LIGHT);
    }
    frame(false, 3);
    if is_excute(agent) {
        HIT_NODE(agent, Hash40::new("snout"), *HIT_STATUS_NORMAL);
    }
}

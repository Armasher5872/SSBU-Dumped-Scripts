    frame(fighter.lua_state_agent, 5.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_szerosuit_rise"));
    }
    frame(fighter.lua_state_agent, 10.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_szerosuit_dash_start"));
    }
    frame(fighter.lua_state_agent, 25.0);
    if macros::is_excute(fighter) {
        macros::STOP_SE(fighter, Hash40::new("se_szerosuit_dash_start"));
        macros::PLAY_SE(fighter, Hash40::new("se_szerosuit_landing01"));
    }
}

#[acmd_script( agent = "trail_hole", script = "game_stay", category = ACMD_GAME, low_priority )]
unsafe fn game_stay(fighter: &mut L2CAgentBase) {
    if WorkModule::is_flag(fighter.module_accessor, *WEAPON_TRAIL_HOLE_INSTANCE_WORK_ID_FLAG_GROUND) {
        for _ in 0..6 {
        if macros::is_excute(fighter) {
            macros::QUAKE(fighter, *CAMERA_QUAKE_KIND_M);
        }
        wait(fighter.lua_state_agent, 3.0);
    }
    frame(fighter.lua_state_agent, 30.0);
    if macros::is_excute(fighter) {
        AttackModule::clear_all(fighter.module_accessor);
        WorkModule::off_flag(fighter.module_accessor, *WEAPON_TRAIL_HOLE_STATUS_WORK_FLAG_PULL);
        WorkModule::on_flag(fighter.module_accessor, *WEAPON_TRAIL_HOLE_STATUS_WORK_FLAG_SEND_ATTACK_END);
    }
}
else{
for _ in 0..6 {
if macros::is_excute(fighter) {
    macros::QUAKE(fighter, *CAMERA_QUAKE_KIND_M);
}
wait(fighter.lua_state_agent, 3.0);
}
frame(fighter.lua_state_agent, 30.0);
if macros::is_excute(fighter) {
AttackModule::clear_all(fighter.module_accessor);
WorkModule::off_flag(fighter.module_accessor, *WEAPON_TRAIL_HOLE_STATUS_WORK_FLAG_PULL);
WorkModule::on_flag(fighter.module_accessor, *WEAPON_TRAIL_HOLE_STATUS_WORK_FLAG_SEND_ATTACK_END);
}
}
}

#[acmd_script( agent = "element_diver", script = "game_ffinalstart" , category = ACMD_GAME , low_priority)]
unsafe fn game_ffinalstart(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if is_excute(fighter) {
        VisibilityModule::set_whole(boma, false);
    }
    frame(lua_state, 1.0);
    if is_excute(fighter) {
        VisibilityModule::set_whole(boma, true);
    }
    frame(lua_state, 5.0);
    if is_excute(fighter) {
        WorkModule::on_flag(boma, *WEAPON_ELEMENT_DIVER_INSTANCE_WORK_ID_FLAG_FINAL_START_GROUND_ENABLE);
    }
    frame(lua_state, 122.0);
    if is_excute(fighter) {
        VisibilityModule::set_whole(boma, false);
    }
}

#[acmd_script( agent = "miiswordsman_chakram", script = "sound_stick", category = ACMD_SOUND, low_priority )]
unsafe fn sound_stick(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::STOP_SE(fighter, Hash40::new("se_miiswordsman_special_c3_s01"));
        macros::STOP_SE(fighter, Hash40::new("se_miiswordsman_special_c3_s04"));
        macros::STOP_SE(fighter, Hash40::new("se_miiswordsman_special_c3_s05"));
    }
    if !WorkModule::is_flag(fighter.module_accessor, *WEAPON_MIISWORDSMAN_CHAKRAM_INSTANCE_WORK_ID_FLAG_FLICK) {
        if macros::is_excute(fighter) {
            macros::PLAY_SE(fighter, Hash40::new("se_miiswordsman_special_c3_s06"));
        }
        else{
        if macros::is_excute(fighter) {
            macros::PLAY_SE(fighter, Hash40::new("se_miiswordsman_special_c3_s02"));
        }
    }
}
}

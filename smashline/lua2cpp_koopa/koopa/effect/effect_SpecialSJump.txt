#[acmd_script( agent = "koopa", script = "effect_specialsjump", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_specialsjump(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 3.0);
    if macros::is_excute(agent) {
        macros::EFFECT_FOLLOW(agent, Hash40::new("koopa_wait_breath"), Hash40::new("top"), 0, 0, 0, 0, 0, 0, 2, true);
    }
    for _ in 0..2 {
    if macros::is_excute(agent) {
        macros::EFFECT_FOLLOW_ALPHA(agent, Hash40::new("koopa_diving"), Hash40::new("rot"), 0, 0, 0, 90, 0, 0, 2, true, 1);
    }
    wait(agent.lua_state_agent, 8.0);
    if macros::is_excute(agent) {
        macros::EFFECT_FOLLOW_ALPHA(agent, Hash40::new("koopa_diving"), Hash40::new("rot"), 5, 0, 0, 90, 0, 0, 1.7, true, 0.8);
    }
    wait(agent.lua_state_agent, 6.0);
}
if macros::is_excute(agent) {
    macros::EFFECT_OFF_KIND(agent, Hash40::new("koopa_wait_breath"), false, false);
}
}

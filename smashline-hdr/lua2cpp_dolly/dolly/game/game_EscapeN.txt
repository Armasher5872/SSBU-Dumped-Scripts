#[acmd_script( agent = "dolly", script = "game_escapen", category = ACMD_GAME, low_priority )]
unsafe fn game_escapen(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 10.0);
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_DOLLY_STATUS_ESCAPE_WORK_FLAG_ATTACK);
    }
    frame(lua_state, 18.0);
    if is_excute(agent) {
        WorkModule::off_flag(boma, *FIGHTER_DOLLY_STATUS_ESCAPE_WORK_FLAG_ATTACK);
    }
}

unsafe extern "C" fn sound_turndash(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 8.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_ike_dash_start"));
        macros::SET_PLAY_INHIVIT(agent, Hash40::new("se_ike_dash_start"), 20);
        macros::SET_PLAY_INHIVIT(agent, Hash40::new("se_ike_step_right_m"), 25);
        macros::SET_PLAY_INHIVIT(agent, Hash40::new("se_ike_step_left_m"), 25);
    }
    wait(agent.lua_state_agent, 8.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_ike_step_right_s"));
    }
    wait(agent.lua_state_agent, 10.0);
    if macros::is_excute(agent) {
        macros::PLAY_STEP(agent, Hash40::new("se_ike_step_left_s"));
    }
}

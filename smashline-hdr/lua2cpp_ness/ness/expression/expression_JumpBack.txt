#[acmd_script( agent = "ness", script = "expression_jumpback", category = ACMD_EXPRESSION, low_priority )]
unsafe fn expression_jumpback(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if IS_FIGHTER_STATUS_KIND(agent, *FIGHTER_STATUS_KIND_GIMMICK_SPRING_JUMP) {
        if is_excute(agent) {
            ControlModule::set_rumble(boma, Hash40::new("rbkind_bounce"), 0, true, *BATTLE_OBJECT_ID_INVALID as u32);
        }
        else{
        if IS_FIGHTER_STATUS_KIND(agent, *FIGHTER_STATUS_KIND_WARPSTAR_JUMP) {
            if is_excute(agent) {
                ControlModule::set_rumble(boma, Hash40::new("rbkind_none"), 0, true, *BATTLE_OBJECT_ID_INVALID as u32);
            }
            else{
            if is_excute(agent) {
                ControlModule::set_rumble(boma, Hash40::new("rbkind_jump"), 0, false, *BATTLE_OBJECT_ID_INVALID as u32);
            }
        }
    }
}
}
}

#[acmd_script( agent = "shizue_fishingrod", script = "game_start", category = ACMD_GAME, low_priority )]
unsafe fn game_start(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    frame(lua_state, 0.0);
    if is_excute(agent) {
        MotionModule::set_rate(boma, 1.25);
    }
    frame(lua_state, 20.0);
    if is_excute(agent) {
        MotionModule::set_rate(boma, 1);
    }
}

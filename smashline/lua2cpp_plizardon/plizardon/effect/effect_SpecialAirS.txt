#[acmd_script( agent = "plizardon", script = "effect_specialairs" , category = ACMD_EFFECT , low_priority)]
unsafe fn effect_specialairs(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 2.0);
    if macros::is_excute(fighter) {
        macros::EFFECT_FOLLOW_FLIP(fighter, Hash40::new("plizardon_flare_blitz"), Hash40::new("plizardon_flare_blitz"), Hash40::new("rot"), 0, 2, 15, 90, 0, 0, 1.3, true, *EF_FLIP_NONE);
        EffectModule::enable_sync_init_pos_last(fighter.module_accessor);
    }
    if macros::is_excute(fighter) {
        macros::BURN_COLOR(fighter, 2, 0.1, 0, 0.7);
    }
    wait(fighter.lua_state_agent, 1.0);
    if macros::is_excute(fighter) {
        macros::BURN_COLOR_FRAME(fighter, 2, 1, 0.2, 0.1, 0);
    }
    wait(fighter.lua_state_agent, 1.0);
    if macros::is_excute(fighter) {
        macros::BURN_COLOR_NORMAL(fighter);
    }
    wait(fighter.lua_state_agent, 1.0);
}

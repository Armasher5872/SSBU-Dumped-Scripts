#[acmd_script( agent = "tantan", script = "sound_turnrun", category = ACMD_SOUND, low_priority )]
unsafe fn sound_turnrun(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 1.0);
    if macros::is_excute(fighter) {
        macros::STOP_SE(fighter, Hash40::new("se_tantan_dash_stop"));
    }
    frame(fighter.lua_state_agent, 3.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_tantan_dash_turn"));
        macros::SET_PLAY_INHIVIT(fighter, Hash40::new("se_tantan_dash_stop"), 20);
    }
}

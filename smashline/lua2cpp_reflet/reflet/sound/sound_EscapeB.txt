    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_reflet_escape"));
    }
    wait(agent.lua_state_agent, 13.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_reflet_step_right_m"));
    }

    }
}

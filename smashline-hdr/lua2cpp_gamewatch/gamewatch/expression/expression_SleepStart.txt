#[acmd_script( agent = "gamewatch", script = "expression_sleepstart", category = ACMD_EXPRESSION, low_priority )]
unsafe fn expression_sleepstart(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        slope!(agent, *MA_MSC_CMD_SLOPE_SLOPE, *SLOPE_STATUS_TOP);
        VisibilityModule::set_int64(boma, hash40("hand") as i64, hash40("hand_hold_lr") as i64);
    }
    wait(lua_state, 9.0);
    if is_excute(agent) {
        VisibilityModule::set_int64(boma, hash40("hand") as i64, hash40("hand_normal") as i64);
    }
}

#[acmd_script( agent = "inkling", script = "sound_turndash" , category = ACMD_SOUND , low_priority)]
unsafe fn sound_turndash(fighter: &mut L2CAgentBase) {
    frame(fighter.lua_state_agent, 3.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_inkling_dash_start"));
        macros::SET_PLAY_INHIVIT(fighter, Hash40::new("se_inkling_dash_start"), 20);
    }
    frame(fighter.lua_state_agent, 14.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_inkling_dash_turn"));
    }
}

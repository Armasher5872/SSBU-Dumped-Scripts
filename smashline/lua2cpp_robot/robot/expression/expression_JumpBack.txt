#[acmd_script( agent = "robot", script = "expression_jumpback", category = ACMD_EXPRESSION, low_priority )]
unsafe fn expression_jumpback(fighter: &mut L2CAgentBase) {
    if macros::IS_FIGHTER_STATUS_KIND(fighter, *FIGHTER_STATUS_KIND_GIMMICK_SPRING_JUMP) {
        if macros::is_excute(fighter) {
            ControlModule::set_rumble(fighter.module_accessor, Hash40::new("rbkind_bounce"), 0, true, *BATTLE_OBJECT_ID_INVALID as u32);
        }
        else{
        if macros::IS_FIGHTER_STATUS_KIND(fighter, *FIGHTER_STATUS_KIND_WARPSTAR_JUMP) {
            if macros::is_excute(fighter) {
                ControlModule::set_rumble(fighter.module_accessor, Hash40::new("rbkind_none"), 0, true, *BATTLE_OBJECT_ID_INVALID as u32);
            }
            else{
            if macros::is_excute(fighter) {
                ControlModule::set_rumble(fighter.module_accessor, Hash40::new("rbkind_vernier"), 0, false, *BATTLE_OBJECT_ID_INVALID as u32);
            }
        }
    }
}
}
}

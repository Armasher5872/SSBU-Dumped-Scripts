    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_dolly_dash_start"));
        macros::SET_PLAY_INHIVIT(fighter, Hash40::new("se_dolly_dash_start"), 20);

        macros::PLAY_SE(fighter, Hash40::new("se_dolly_step_left_ll"));
    }

    }
}

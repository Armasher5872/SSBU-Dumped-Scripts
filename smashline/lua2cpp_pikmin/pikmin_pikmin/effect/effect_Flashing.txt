        macros::FLASH(fighter, 1, 1, 1, 0.588);
    }
    wait(fighter.lua_state_agent, 1.0);

        macros::FLASH_FRM(fighter, 5, 0, 0, 0, 0.392);
    }

        macros::COL_NORMAL(fighter);
    }

}

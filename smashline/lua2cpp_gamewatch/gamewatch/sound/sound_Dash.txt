#[acmd_script( agent = "gamewatch", script = "sound_dash", category = ACMD_SOUND, low_priority )]
unsafe fn sound_dash(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 2.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_gamewatch_dash_start"));
        macros::SET_PLAY_INHIVIT(agent, Hash40::new("se_gamewatch_dash_start"), 20);
    }
    wait(agent.lua_state_agent, 2.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_gamewatch_step_right_m"));
    }
    wait(agent.lua_state_agent, 3.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_gamewatch_step_left_m"));
    }
}

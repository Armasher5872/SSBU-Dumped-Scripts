#[acmd_script( agent = "szerosuit", script = "sound_slipescapeb", category = ACMD_SOUND, low_priority )]
unsafe fn sound_slipescapeb(agent: &mut L2CAgentBase) {
    frame(agent.lua_state_agent, 5.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_szerosuit_rise"));
    }
    frame(agent.lua_state_agent, 10.0);
    if macros::is_excute(agent) {
        macros::PLAY_SE(agent, Hash40::new("se_szerosuit_dash_start"));
    }
    frame(agent.lua_state_agent, 25.0);
    if macros::is_excute(agent) {
        macros::STOP_SE(agent, Hash40::new("se_szerosuit_dash_start"));
        macros::PLAY_SE(agent, Hash40::new("se_szerosuit_landing01"));
    }
}

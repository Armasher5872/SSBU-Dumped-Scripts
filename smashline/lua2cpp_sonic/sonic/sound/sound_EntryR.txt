#[acmd_script( agent = "sonic", script = "sound_entryr", category = ACMD_SOUND, low_priority)]
unsafe fn sound_entryr(fighter: &mut L2CAgentBase) {
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_sonic_rounddash"));
    }
    wait(fighter.lua_state_agent, 18.0);
    if macros::is_excute(fighter) {
        macros::PLAY_SE(fighter, Hash40::new("se_sonic_landing03"));
    }
    wait(fighter.lua_state_agent, 40.0);
    if macros::is_excute(fighter) {
        macros::STOP_SE(fighter, Hash40::new("se_sonic_rounddash"));
        macros::PLAY_SE(fighter, Hash40::new("se_sonic_landing01"));
    }
}

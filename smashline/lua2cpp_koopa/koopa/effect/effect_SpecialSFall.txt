#[acmd_script( agent = "koopa", script = "effect_specialsfall", category = ACMD_EFFECT, low_priority )]
unsafe fn effect_specialsfall(agent: &mut L2CAgentBase) {
    if macros::is_excute(agent) {
        macros::EFFECT_FOLLOW_COLOR(agent, Hash40::new("koopa_atk_fall"), Hash40::new("top"), 0, 10, 2, 0, 0, 0, 1, true, 1, 0.902, 0.49);
    }
    for _ in 0..5 {
    if macros::is_excute(agent) {
        macros::EFFECT_FLIP(agent, Hash40::new("sys_spin_wind"), Hash40::new("sys_spin_wind"), Hash40::new("top"), 0, 6, 0, 180, 0, 0, 1.8, 0, 0, 0, 0, 0, 0, true, *EF_FLIP_YZ);
    }
    wait(agent.lua_state_agent, 3.0);
    if macros::is_excute(agent) {
        macros::EFFECT_FLIP(agent, Hash40::new("sys_spin_wind"), Hash40::new("sys_spin_wind"), Hash40::new("top"), 0, -1, 0, 180, -60, 0, 1.7, 0, 0, 0, 0, 0, 0, true, *EF_FLIP_YZ);
    }
    wait(agent.lua_state_agent, 3.0);
}

    frame(agent.lua_state_agent, 3.0);
    if macros::is_excute(agent) {
        JostleModule::set_status(agent.module_accessor, false);
    }
    frame(agent.lua_state_agent, 6.0);
    if macros::is_excute(agent) {

        AttackModule::clear_all(agent.module_accessor);

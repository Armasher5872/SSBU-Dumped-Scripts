#[acmd_script( agent = "kamui", script = "game_jumpfrontmini", category = ACMD_GAME, low_priority )]
unsafe fn game_jumpfrontmini(agent: &mut L2CAgentBase) {
    let lua_state = agent.lua_state_agent;
    let boma = agent.boma();
    if is_excute(agent) {
        WorkModule::on_flag(boma, *FIGHTER_KAMUI_INSTANCE_WORK_ID_FLAG_SPECIAL_S_DISABLE_FOR_JUMP);
    }
    frame(lua_state, 4.0);
    if is_excute(agent) {
        WorkModule::off_flag(boma, *FIGHTER_KAMUI_INSTANCE_WORK_ID_FLAG_SPECIAL_S_DISABLE_FOR_JUMP);
    }
}

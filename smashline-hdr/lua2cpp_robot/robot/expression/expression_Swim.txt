#[acmd_script( agent = "robot", script = "expression_swim", category = ACMD_EXPRESSION, low_priority )]
unsafe fn expression_swim(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if is_excute(fighter) {
        ItemModule::set_have_item_visibility(boma, false);
    }
    if is_excute(fighter) {
        ControlModule::set_rumble(boma, Hash40::new("rbkind_walk"), 0, false, *BATTLE_OBJECT_ID_INVALID as u32);
    }
    wait(lua_state, 4.0);
}
